```c++
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    while(n--) {
        list<char> L;
        list<char>::iterator t;
        string s;
        cin >> s;
        for (char c:s){
            //처음 list가 비어있을 때 기호는 무시한다.
            if (L.empty()) {
                if (c == '<' || c == '>' || c == '-') continue;
                else {
                    //값이 채워지고 iterator를 초기화한다.
                    L.push_front(c);
                    t = L.begin();
                }
            }
            else if (c == '<') {
                if (t != L.begin()) --t;
            }
            else if (c == '>') {
                if (t != L.end()) {
                    t++;
                    if (t == L.end()) t--;
                }
            }
            else if (c == '-') {
                t = L.erase(t);
                t--;
            }
            else {
                t++;
                L.insert(t,c);
                t--;
            }
        }
        for (auto i : L) cout << i;
    }

    return 0;
}
```
런타임 에러 발생, 이유를 모르겠음.

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  int n;

  cin >> n;

  for (int i = 0; i < n; i++) {
    list<char> L = {};
    string s;
    auto p = L.begin();

    cin >> s;
    for (auto c : s) {
      if (c == '<') {
        if (p != L.begin()) p--;
      }
      else if (c == '>') {
        if (p != L.end()) p++;
      }
      else if (c == '-') {
        if (p != L.begin()) {
          p--;
          p = L.erase(p);
        }
      }
      else
        L.insert(p, c);      
    }
    for (auto c : L) cout << c;
    cout << '\n';
  }
}
```
빈 list가 어떻게 작동하는지 몰라서 답의 논리흐름을 따르지 못했음
