틀린이유: char형을 곧바로 배열 인덱스에 대입하여 오류 발생
c-'a'를 통해 int형으로 바꾸면 정상적으로 실행될 것이라 예상
```c++
#include <iostream>
#include <string>

using namespace std;
int main()
{
    int n;
    cin >> n;
    string a, b;
    for (int i = 0; i != n; ++i) {
        int arr[123] = {0,};
        bool check = 1;
        cin >> a >> b;
        for (char c1:a) {
            arr[c1]++;
        }
        for (char c2:b) {
            if (arr[c2] != 1) {
                cout << "Impossible" << '\n';
                check = 0;
                break;
            }
        }
        if (check) cout << "Possible" << '\n';
    }

    return 0;
}
```
그럼에도 틀렸음. 이유를 찾아봤을 때 중복된 문자로 문자열이 만들어질 때 오류가 발생함을 알게됨.
따라서 단순히 배열 값이 1일 때를 확인하는 것이 아니라, 다른 방식이 필요함.
따라서 답처럼 처음 문자열에서 문자마다 1씩 올리고, 다음 문자열에서 문자마다 1씩 내리는 것이 해결책이 됨.

```
#include <iostream>
#include <string>

using namespace std;
int main()
{
    int n;
    cin >> n;
    string a, b;
    for (int i = 0; i != n; ++i) {
        int arr[123] = {0,};
        bool check = 1;
        cin >> a >> b;
        for (char c1:a) {
            arr[c1 - 'a']++;
        }
        for (char c2:b) {
            if (arr[c2 - 'a'] != 1) {
                cout << "Impossible" << '\n';
                check = 0;
                break;
            }
        }
        if (check) cout << "Possible" << '\n';
    }

    return 0;
}
```

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);

  int N;
  cin >> N;
  while (N--) {
    int a[26] = {}; // 각 문자의 개수를 저장하는 배열
    string s1, s2;
    cin >> s1 >> s2;

    for (auto c : s1) a[c-'a']++; // 첫 번째 문자열의 각 문자는 개수+1
    for (auto c : s2) a[c-'a']--; // 두 번째 문자열의 각 문자는 개수-1

    // 0이 아닌 배열의 요소가 있을 경우, 개수가 다른 문자가 존재하므로 false
    bool isPossible = true;
    // 중괄호가 없어도 문제는 없으나 가독성을 위해 삽입
    for (int i : a){
      if (i != 0) isPossible = false;
    }

    if(isPossible) cout << "Possible\n";
    else cout << "Impossible\n";
  }
}
```
